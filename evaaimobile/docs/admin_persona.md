## **🗄️ НОВАЯ СТРУКТУРА БАЗЫ ДАННЫХ**

sql

```
-- Персонажи (главная таблица)
CREATE TABLE characters (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(50) NOT NULL,
    display_name VARCHAR(50),
    age INTEGER,
    personality_type VARCHAR(50),
    biography TEXT,
    secret TEXT,
    avatar_url VARCHAR(500),
    voice_provider VARCHAR(50) DEFAULT 'elevenlabs',
    voice_id VARCHAR(100),
    voice_settings JSONB DEFAULT '{}',
    status VARCHAR(20) DEFAULT 'active',
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Слои (привязаны к персонажу)
CREATE TABLE layers (
    id SERIAL PRIMARY KEY,
    character_id UUID REFERENCES characters(id) ON DELETE CASCADE,
    layer_number INTEGER NOT NULL,
    name VARCHAR(100),
    min_trust_score INTEGER DEFAULT 0,
    max_trust_score INTEGER DEFAULT 1000,
    requirements JSONB DEFAULT '{}',  -- {night_conversations: 2, min_days: 3}
    initiator_prompt TEXT,
    system_prompt_override TEXT,
    is_erotic BOOLEAN DEFAULT FALSE,
    UNIQUE(character_id, layer_number)
);

-- Контент (привязан к слоям персонажа)
CREATE TABLE content (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    character_id UUID REFERENCES characters(id) ON DELETE CASCADE,
    layer_id INTEGER REFERENCES layers(id) ON DELETE CASCADE,
    content_type VARCHAR(20), -- photo, video, audio
    url VARCHAR(500),
    thumbnail_url VARCHAR(500),
    description TEXT,
    tags TEXT[],
    is_erotic BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Системные промпты (версионирование)
CREATE TABLE character_prompts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    character_id UUID REFERENCES characters(id) ON DELETE CASCADE,
    prompt_type VARCHAR(50), -- 'system', 'layer_override'
    layer_id INTEGER, -- для override, NULL для system
    content TEXT NOT NULL,
    version INTEGER,
    created_by UUID, -- admin user id
    created_at TIMESTAMP DEFAULT NOW(),
    is_active BOOLEAN DEFAULT FALSE
);
```

## **🚀 API ENDPOINTS (НОВАЯ СТРУКТУРА)**

python

```
# ВСЕ ЭНДПОИНТЫ ГРУППИРУЮТСЯ ВОКРУГ ПЕРСОНАЖА

# Базовые CRUD персонажа
GET    /api/admin/characters                     # список персонажей
POST   /api/admin/characters                     # создание
GET    /api/admin/characters/{id}                # детали персонажа
PUT    /api/admin/characters/{id}                # обновление
DELETE /api/admin/characters/{id}                # удаление

# Слои персонажа
GET    /api/admin/characters/{id}/layers         # все слои
POST   /api/admin/characters/{id}/layers         # создать слой
PUT    /api/admin/characters/{id}/layers/{lid}   # обновить слой
DELETE /api/admin/characters/{id}/layers/{lid}   # удалить слой

# Контент персонажа
GET    /api/admin/characters/{id}/content        # вся медиатека
POST   /api/admin/characters/{id}/content/upload # загрузить контент
DELETE /api/admin/characters/{id}/content/{cid}  # удалить контент

# Промпты персонажа
GET    /api/admin/characters/{id}/prompts/system # системный промпт
PUT    /api/admin/characters/{id}/prompts/system # обновить системный
GET    /api/admin/characters/{id}/prompts/layers # override по слоям
PUT    /api/admin/characters/{id}/prompts/layers/{lid} # обновить override
GET    /api/admin/characters/{id}/prompts/history # история версий

# Голос персонажа
GET    /api/admin/characters/{id}/voice          # настройки голоса
PUT    /api/admin/characters/{id}/voice          # обновить голос
POST   /api/admin/characters/{id}/voice/test     # тестовая генерация
```
